284 void merge(double arr[], int l, int m, int r)
285 {
286 int i, j, k;
287 int n1 = m - l + 1;
288 int n2 = r - m;
289
290 /* create temp arrays */
291 double L[n1], R[n2];
292
293 /* Copy data to temp arrays L[] and R[] */
294 for (i = 0; i < n1; i++)
295 L[i] = arr[l + i];
296 for (j = 0; j < n2; j++)
297 R[j] = arr[m + 1 + j];
298
299 /* Merge the temp arrays back into arr[l..r]*/
300 i = 0; // Initial index of first subarray
301 j = 0; // Initial index of second subarray
302 k = l; // Initial index of merged subarray
303 while (i < n1 && j < n2)
304 {
305 if (L[i] <= R[j])
306 {
307 arr[k] = L[i];
308 i++;
309 }
310 else
311 {
312 arr[k] = R[j];
313 j++;
314 }
315 k++;
316 }
317
318 /* Copy the remaining elements of L[], if there
319 are any */
320 while (i < n1)
321 {
322 arr[k] = L[i];
323 i++;
324 k++;
325 }
326
327 /* Copy the remaining elements of R[], if there
328 are any */
329 while (j < n2)
330 {
331 arr[k] = R[j];
332 j++;
333 k++;
334 }
335 }

343 void mergeSort(double arr[], double l, double r)
344 {
345 if (l < r)
